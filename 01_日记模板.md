```dataviewjs
const baseId = "SPORT";
const options = ["🏊🏻‍♀️", "🏸", "❌"];
const index = dv.current().file.name.replace(".md", "");
const id = `${baseId}_${index}`;
const saved = localStorage.getItem(id) || "";
const container = this.container.createEl("div");
const select = container.createEl("select", { attr: { id: "my_dropdown"  } });
const placeholder = select.createEl("option", {text: "今日运动状态",attr: { value: "", disabled: true }});
if (!saved) placeholder.selected = true;
for (const opt of options) {
  const o = select.createEl("option", { text: opt, attr: { value: opt } });
  if (saved === opt) o.selected = true;
}
select.addEventListener("change", e => {
  localStorage.setItem(id, e.target.value);
});
```
